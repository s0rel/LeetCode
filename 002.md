## 2. Add Two Numbers

### Description
You are given two non-empty linked lists representing two non-negative integers. The digits are stored in reverse order and each of their nodes contain a single digit. Add the two numbers and return it as a linked list. You may assume the two numbers do not contain any leading zero, except the number 0 itself.

### Example
Input: (2 -> 4 -> 3) + (5 -> 6 -> 4)     
Output: 7 -> 0 -> 8     
Explanation: 342 + 465 = 807.     

### Solution
Java，紧凑的写法
```Java
class Solution {
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
        int count = 0;
        int value;
        ListNode origin = new ListNode(-1);
        ListNode result = new ListNode(-1);
        origin = result;
        while (l1 != null || l2 != null) {
            value = (count + (l1 == null ? 0 : l1.val) + (l2 == null ? 0 : l2.val)) % 10;
            count = (count + (l1 == null ? 0 : l1.val) + (l2 == null ? 0 : l2.val)) / 10;
            result.next = new ListNode(value);
            result = result.next;
            if (l1 != null)
                l1 = l1.next;
            if (l2 != null)
                l2 = l2.next;
        }
        if (count == 1) {
            result.next = new ListNode(count);
            result = result.next;
        }
        return origin.next;
    }
}
```
更清晰的写法
```Java
 class Solution {
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
        ListNode origin = new ListNode(-1);
        ListNode result = origin;
        int flag = 0, sum = 0;

        while (l1 != null && l2 != null) {
            sum = (l1.val + l2.val + flag) % 10;
            flag = (l1.val + l2.val + flag) / 10;
            ListNode temp = new ListNode(sum);
            result.next = temp;
            result = result.next;
            l1 = l1.next;
            l2 = l2.next;
        }

        if (l1 == null && l2 == null) {
            if (flag != 0) {
                ListNode temp = new ListNode(flag);
                result.next = temp;
                result = result.next;
            }
        }

        if (l1 != null) {
            while (l1 != null) {
                sum = (l1.val + flag) % 10;
                flag = (l1.val + flag) / 10;
                ListNode temp = new ListNode(sum);
                result.next = temp;
                result = result.next;
                if (l1.next == null) {
                    if (flag != 0) {
                        ListNode tmp = new ListNode(flag);
                        result.next = tmp;
                        result = result.next;
                    }
                }
                l1 = l1.next;
            }
        }

        if (l2 != null) {
            while (l2 != null) {
                sum = (l2.val + flag) % 10;
                flag = (l2.val + flag) / 10;
                ListNode temp = new ListNode(sum);
                result.next = temp;
                result = result.next;
                if (l2.next == null) {
                    if (flag != 0) {
                        ListNode tmp = new ListNode(flag);
                        result.next = tmp;
                        result = result.next;
                    }
                }
                l2 = l2.next;
            }
        }
        return origin.next;
    }
}
```
其中，ListNode的定义如下
```Java
public class ListNode {
    int val;
    ListNode next;

    ListNode(int x) {
        val = x;
    }
}
```

public class AddTwoNumbers {

        Solution sol = new Solution();
        ListNode h1 = new ListNode(1);
        ListNode node1 = new ListNode(4);
        ListNode node2 = new ListNode(3);
        h1.next = node1;
        node1.next = node2;

        ListNode h2 = new ListNode(9);
        ListNode node3 = new ListNode(9);
        ListNode node4 = new ListNode(4);
        h2.next = node3;
        node3.next = node4;

        ListNode res = sol.addTwoNumbers(h1, h2);
        while (res != null) {
            System.out.println(res.val);
            res = res.next;
        }
    }
}
